[docs]def reduce_dic(dico, remove):
    """ merges key, values of a dictionary
    
    Parameters
    ----------
    dico
    remove

    Returns
    -------

    """
    res = ''
    for key, value in dico.items():
        if key not in remove:
            res += ' ' + key + ' ' + value

    return res
